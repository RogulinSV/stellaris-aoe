### Crystalline Worlds survey

ship_event = {
    id = aoe_planet_actions.1
    title = aoe_planet_actions.1.title
    desc = aoe_planet_actions.1.desc
    picture = GFX_aoe_crystalline_worlds
    show_sound = event_laboratory_sound
    is_triggered_only = yes
    hide_window = no
    location = from

    trigger = {
        owner = {
            NOT = {
                has_event_chain = "crystalline_worlds_chain"
                has_country_flag = "crystalline_worlds_surveyed"
                has_country_flag = "crystalline_worlds_discovered"
                has_ethic = "ethic_gestalt_consciousness"
            }
        }
        from = {
            OR = {
                is_planet_class = "pc_crystalline"
                is_planet_class = "pc_crystalline_uninhabitable"
                is_planet_class = "pc_crystalline_gas_giant"
            }
        }
    }

    immediate = {
        owner = {
            set_timed_country_flag = {
                flag = "crystalline_worlds_discovered"
                days = 720
            }
        }
    }

    option = {
        name = aoe_planet_actions.1.a
        custom_tooltip = aoe_planet_actions.1.a.tooltip
        begin_event_chain = {
            event_chain = "crystalline_worlds_chain"
            target = owner
        }
    }
    option = {
        name = aoe_planet_actions.1.b
        custom_tooltip = aoe_planet_actions.1.b.tooltip
        leader = {
            add_experience = 25
        }
    }
}

ship_event = {
    id = aoe_planet_actions.2
    is_triggered_only = yes
    hide_window = yes

    trigger = {
        owner = {
            has_event_chain = "crystalline_worlds_chain"
        }
        from = {
            OR = {
                is_planet_class = "pc_crystalline"
                is_planet_class = "pc_crystalline_uninhabitable"
                is_planet_class = "pc_crystalline_gas_giant"
            }
        }
    }

    immediate = {
        if = {
            limit = {
                from = {
                    is_planet_class = "pc_crystalline"
                }
                owner = {
                    NOT = {
                        has_completed_event_chain_counter = {
                            event_chain = "crystalline_worlds_chain"
                            counter = "crystalline_worlds_surveyed"
                        }
                    }
                }
            }
            owner = {
                add_event_chain_counter = {
                    event_chain = "crystalline_worlds_chain"
                    counter = "crystalline_worlds_surveyed"
                    amount = 1
                }
            }
        }
        else_if = {
            limit = {
                from = {
                    is_planet_class = "pc_crystalline_uninhabitable"
                }
                owner = {
                    NOT = {
                        has_completed_event_chain_counter = {
                            event_chain = "crystalline_worlds_chain"
                            counter = "crystalline_uninhabitable_worlds_surveyed"
                        }
                    }
                }
            }
            owner = {
                add_event_chain_counter = {
                    event_chain = "crystalline_worlds_chain"
                    counter = "crystalline_uninhabitable_worlds_surveyed"
                    amount = 1
                }
            }
        }
        else_if = {
            limit = {
                from = {
                    is_planet_class = "pc_crystalline_gas_giant"
                }
                owner = {
                    NOT = {
                        has_completed_event_chain_counter = {
                            event_chain = "crystalline_worlds_chain"
                            counter = "crystalline_gas_giant_worlds_surveyed"
                        }
                    }
                }
            }
            owner = {
                add_event_chain_counter = {
                    event_chain = "crystalline_worlds_chain"
                    counter = "crystalline_gas_giant_worlds_surveyed"
                    amount = 1
                }
            }
        }

        if = {
            limit = {
                owner = {
                    has_completed_event_chain_counter = {
                        event_chain = "crystalline_worlds_chain"
                        counter = "crystalline_worlds_surveyed"
                    }
                    has_completed_event_chain_counter = {
                        event_chain = "crystalline_worlds_chain"
                        counter = "crystalline_uninhabitable_worlds_surveyed"
                    }
                    has_completed_event_chain_counter = {
                        event_chain = "crystalline_worlds_chain"
                        counter = "crystalline_gas_giant_worlds_surveyed"
                    }
                }
            }
            owner = {
                country_event = { id = aoe_planet_actions.3 }
            }
        }
    }
}
