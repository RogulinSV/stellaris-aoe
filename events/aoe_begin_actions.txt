    show_sound = event_sensor_ping
    hide_window = no
    is_triggered_only = yes

    trigger = {
        # has_global_flag = planetarydiversity
    }

    option = {
        name = aoe_begin_actions.10.a
        custom_tooltip = aoe_begin_actions.10.a.tooltip
        hidden_effect = {
            solar_system = {
                spawn_system = {
                    min_distance >= 20
                    max_distance <= 50
                    initializer = "anthem_of_eternity_neighbor_a"
                }
                spawn_system = {
                    min_distance >= 30
                    max_distance <= 50
                    initializer = "anthem_of_eternity_neighbor_b"
                }
                spawn_system = {
                    min_distance >= 30
                    max_distance <= 50
                    initializer = "anthem_of_eternity_neighbor_c"
                }
                spawn_system = {
                    min_distance >= 40
                    max_distance <= 60
                    initializer = "anthem_of_eternity_neighbor_d"
                }
            }
            every_system = {
                limit = {
                    has_star_flag = "flags_aoe_neighbor_world"
                }
                while = {
                    count = 3
                    closest_system = {
                        min_steps = 1
                        max_steps = 3
                        limit = {
                            NOR = {
                                is_same_value = prev
                                has_hyperlane_to = prev
                            }
                        }
                        add_hyperlane = {
                            from = this
                            to = prev
                        }

                        log = "aoe_begin_actions.10: System «[This.GetName]» was connected with «[Prev.GetName]» system"
                    }
                }
            }
        }
    }
    option = {
        name = aoe_begin_actions.decline
    }
}
