namespace = aoe_battle_actions

country_event = {
    id = aoe_battle_actions.1
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        if = {
            limit = {
                check_variable = {
                    which = "counter_space_battles_total_won"
                    value < 1
                }
            }
            set_variable = {
                which = "counter_space_battles_total_won"
                value = 1
            }
        }
        else = {
            change_variable = {
                which = "counter_space_battles_total_won"
                value = 1
            }
        }
    }
}
