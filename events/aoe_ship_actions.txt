namespace = aoe_ship_actions

### Ship building based events

ship_event = {
    id = aoe_ship_actions.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_ship_size = "corvette"
    }

    immediate = {
        owner = {
            if = {
                limit = {
                    check_variable = {
                        which = "counter_ship_corvettes_built"
                        value < 1
                    }
                }
                set_variable = {
                    which = "counter_ship_corvettes_built"
                    value = 1
                }
                else = {
                    change_variable = {
                        which = "counter_ship_corvettes_built"
                        value = 1
                    }
                }
            }
        }
    }
}

### Strategic resource specific ship events

ship_event = {
    id = aoe_ship_actions.10
    title = aoe_ship_actions.10.title
    desc = aoe_ship_actions.10.desc
    picture = GFX_aoe_metallic_resources_discovered
    show_sound = event_laboratory_sound
    is_triggered_only = yes
    hide_window = no
    location = from

    immediate = {
        leader = {
            add_experience = 50
        }
    }

    option = {
        name = aoe_ship_actions.10.a
        custom_tooltip = aoe_ship_actions.10.a.tooltip
        trigger = {
            owner = {
                has_technology = "tech_mine_mithril"
            }
            fromfrom = {
                has_mithril_resource = yes
            }
        }
        owner = {
            add_resource = {
                mithril = @tier2materialmax
            }
        }
    }
    option = {
        name = aoe_ship_actions.10.b
        custom_tooltip = aoe_ship_actions.10.b.tooltip
        trigger = {
            owner = {
                NOT = {
                    has_technology = "tech_mine_mithril"
                    has_special_project = "aoe_metallic_resources_anomaly_project"
                }
            }
            fromfrom = {
                has_mithril_resource = yes
            }
        }
        hidden_effect = {
            fromfrom = {
                enable_special_project = {
                    name = "aoe_metallic_resources_anomaly_project"
                    location = this
                    owner = root.owner
                }
            }
        }
    }
    option = {
        name = aoe_ship_actions.10.c
        custom_tooltip = aoe_ship_actions.10.c.tooltip
    }
}
