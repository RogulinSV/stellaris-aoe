### Propaganda Center

building_propaganda_center_1 = {
    capital = no
    can_build = yes
    can_demolish = yes
    can_be_ruined = yes
    can_be_disabled = yes
    base_buildtime = 400
    base_cap_amount = 1
    category = unity

    upgrades = {
        "building_propaganda_center_2"
    }

    potential = {
        is_colonizeable_planet = yes
        has_upgraded_capital = yes
        exists = owner
        owner = { 
            is_gestalt = no
            is_fallen_empire = no
        }
    }

    allow = {
        is_sovereign_colony = yes
    }

    destroy_trigger = {
        exists = owner
        OR = {
            is_sovereign_colony = no
            owner = { 
                OR = {
                    is_gestalt = yes
                    is_fallen_empire = yes
                }
            }
        }
    }

    show_tech_unlock_if = {
        is_gestalt = no
        is_fallen_empire = no
    }

    resources = {
        category = planet_buildings
        cost = {
            energy = 1000
            nevarrit = 500
        }
        upkeep = {
            nevarrit = 2
            minerals = 4
        }
    }

    planet_modifier = {
        planet_jobs_unity_produces_mult = 0.1
    }

    country_modifier = {
        country_unity_produces_mult = 0.05
    }

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_gestalt = no
                is_fallen_empire = no
            }
        }
        modifier = {
            job_unity_propagandist_add = 2
        }
    }

    triggered_desc = {
        trigger = {
            exists = owner
            owner = {
                is_gestalt = no
                is_fallen_empire = no
            }
        }
        text = "job_unity_propagandist_effect_desc"
    }

    ai_weight = {
        weight = 1

        modifier = {
            factor = 0
            owner = {
                has_monthly_income = {
                    resource = nevarrit
                    value < 3
                }
            }
        }
        modifier = {
            factor = 0
            owner = {
                has_monthly_income = {
                    resource = minerals
                    value < 5
                }
            }
        }
    }

    prerequisites = {
        "tech_terrestrial_propaganda"
    }
}
