        }
    }

    triggered_desc = {
        trigger = {
            exists = owner
            owner = {
                is_gestalt = no

            }
        }
        text = "job_vice_squad_effect_desc"
    }

    ai_weight = {
        weight = 1

        modifier = {
            factor = 0
            owner = {
                has_monthly_income = {
                    resource = energy
                    value < 2
                }
            }
        }
        modifier = {
            factor = 0
            owner = {
                has_monthly_income = {
                    resource = minerals
                    value < 2
                }
            }
        }
        modifier = {
            factor = 0
            owner = {
                has_monthly_income = {
                    resource = orihalk
                    value < 2
                }
            }
        }
        modifier = {
            factor = 1000
            OR = {
                has_planet_modifier = "planet_stability_high"
                has_planet_modifier = "planet_stability_no_happiness_high"
                has_planet_modifier = "planet_crime"
                has_planet_modifier = "planet_crime_no_happiness"
                planet_crime > 20

        modifier = {
            factor = 100
            OR = {
                has_planet_modifier = "planet_stability_low"

            }
        }
        modifier = {
            factor = 10
            planet_crime > 2
        }
        modifier = {
            factor = 0
            planet_crime < 1
        }
    }
