replicant_fleet_patrol_effect = {
    fleet_event = {
        id = aoe_pirate_actions.20
        days = 30
        random = 25
    }
}

replicant_fleet_repair_effect = {
    if = {
        limit = {
            exists = owner
        }
        owner = {
            every_owned_ship = {
                limit = {
                    exists = fleet
                    fleet = {
                        is_same_value = prevprev
                    }
                }
                set_disabled = yes
            }
        }
    }
    fleet_event = {
        id = aoe_pirate_actions.21
        days = 90
    }
}

replicant_fleet_destroy_effect = {
    fleet_event = {
        id = aoe_pirate_actions.22
        days = 50
        random = 25
    }
}

replicant_fleet_dismiss_effect = {
    if = {
        limit = {
            exists = leader
        }
        leader = {
            kill_leader = {
                show_notification = no
            }
        }
    }
    if = {
        limit = {
            exists = owner
        }
        owner = {
            every_owned_ship = {
                limit = {
                    exists = fleet
                    fleet = {
                        is_same_value = prevprev
                    }
                }
                set_disabled = yes
            }
        }
    }
    fleet_event = {
        id = aoe_pirate_actions.23
        days = 30
    }
}
