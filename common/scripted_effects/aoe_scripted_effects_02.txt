# Habitable planets
effects_add_rich_resources_habitable_planet = {
    switch = {
        trigger = planet_size
        50 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 47
            }
        }
        45 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 42
            }
        }
        40 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 37
            }
        }
        35 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 32
            }
        }
        30 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 27
            }
        }
        25 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 22
            }
        }
        20 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 17
            }
        }
        15 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 12
            }
        }
        10 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 7
            }
        }
        0 > {
            set_variable = {
                which = get_planet_deposits_count
                value = 0
            }
        }
    }

    while = {
        limit = {
            check_variable = {
                which = get_planet_deposits_count
                value > 0
            }
        }

        effects_add_planet_deposit = yes

        change_variable = {
            which = get_planet_deposits_count
            value = -1
        }
    }
}
