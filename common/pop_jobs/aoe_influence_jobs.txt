### Influence Jobs

nexus_agent = {
    category = specialist
    condition_string = SPECIALIST_JOB_TRIGGER
    building_icon = "building_nexus_agency_4"
    clothes_texture_index = 3

    possible_pre_triggers = {
        has_owner = yes
        is_enslaved = no
        is_being_purged = no
        is_being_assimilated = no
    }

    possible = {
        complex_specialist_job_check_trigger = yes
        hidden_trigger = { exists = owner }
        owner = {
            is_gestalt = no
            is_fallen_empire = no
        }
    }

    resources = {
        category = planet_enforcers
        upkeep = {
            unity = 1
        }
        upkeep = {
            trigger = {
                exists = planet
                planet = {
                    NOT = {
                        is_sovereign_colony = yes
                        has_orbital_bombardment = no
                    }
                }
            }
            minerals = 2
        }
        upkeep = {
            trigger = {
                exists = pop
                pop = {
                    is_pop_purged = yes
                }
            }
            minerals = 4
        }
        produces = {
            influence = 1
        }
    }

    planet_modifier = {
        pop_cat_ruler_happiness = 0.05
    }

    weight = {
        weight = @specialist_job_weight

        modifier = {
            factor = 1.5
            has_trait = "trait_conformists"
        }
        modifier = {
            factor = 1.3
            has_trait = "trait_communal"
        }
        modifier = {
            factor = 1.1
            has_trait = "trait_ingenious"
        }
        modifier = {
            factor = 0.5
            has_trait = "trait_intelligent"
        }
        modifier = {
            factor = 0.1
            can_take_servant_job = yes
        }
        modifier = {
            factor = 0.5
            planet = {
                OR = {
                    has_orbital_bombardment = yes
                    has_ground_combat = yes
                }
            }
        }
        modifier = {
            factor = 0.01
            is_pop_purged = yes
        }
        modifier = {
            factor = 0.01
            planet.controller = {
                OR = {
                    is_country_type = swarm
                    is_country_type = ai_empire
                }
            }
        }
    }
}
