special_project = {
    key = "aoe_crystalline_adaptation_project"
    cost = @project_cost
    timelimit = -1
    location = no
    tech_department = society_technology
    picture = GFX_aoe_crystalline_adaptation_project
    icon = "gfx/interface/icons/situation_log/situation_log_planet.dds"

    event_scope = planet_event

    abort_trigger = {
        from = {
            OR = {
                has_ground_combat = yes
                has_orbital_bombardment = yes

                    is_controlled_by = root
                    owner = {
                        is_same_value = root
                    }
                }
            }
        }
    }

    on_success = {
        capital_scope = {
            modify_species = {
                add_trait = "trait_crystalline_species"
            }
        }
    }
}
