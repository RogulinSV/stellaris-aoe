                    tech = "tech_mine_mithril"
                }
            }
        }

                    has_orihalk_resource = yes
                }

                    }
                }
            }
            owner = {
                give_technology = {
                    tech = "tech_mine_orihalk"
                }
            }
        }
        else_if = {
            limit = {
                from = {
                    has_yuranic_resource = yes
                }

                    NOT = {
                        has_technology = "tech_mine_yuranic"
                    }
                }
            }
            owner = {
                give_technology = { 
                    tech = "tech_mine_yuranic"
                }
            }
        }

            limit = {
                from = {

                    }
                }

                    max = @tier1researchmax

            }
        }
