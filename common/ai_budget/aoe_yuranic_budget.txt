    category = buffer

}

yuranic_expenditure_starbases = {
    resource = yuranic
    type = expenditure
    category = starbases

    potential = {
        NOT = {
            is_country_type = fallen_empire
        }
    }

    fraction = {
        weight = 0.1
    }
}

yuranic_expenditure_starbases_expand = {
    resource = yuranic
    type = expenditure
    category = starbases

    potential = {
        has_ai_expansion_plan = yes
        highest_threat < 50
        NOT = {
            is_country_type = fallen_empire
            is_country_type = awakened_fallen_empire
        }
    }

    fraction = {
        weight = 0.1
    }
}
