d_melted_streams = {
    icon = d_melted_streams
    time = 400

    is_for_colonizeable = yes
    category = deposit_cat_blockers

    blocker_swap_types = {
        d_fertile_lands
        d_great_river
    }

    potential = {
        planet_size >= 10
        is_dry = yes
    }

    resources = {
        category = deposit_blockers
        cost = {
            energy = 500
            minerals = 300
        }
    }

    planet_modifier = { 
        planet_max_districts_add = -1
    }

    on_cleared = {
        hidden_effect = {
            if = {
                limit = {
                    exists = owner
                    owner = {
                        has_technology = "tech_advanced_terraforming"
                    }
                }
                owner = {
                    add_resource = {
                        energy = 200
                        minerals = 300
                    }
                }
            }
            else_if = {
                limit = {
                    exists = owner
                    owner = {
                        has_technology = "tech_supreme_terraforming"
                    }
                }
                owner = {
                    add_resource = {
                        energy = 300
                        minerals = 450
                    }
                }
            }
            else_if = {
                limit = {
                    exists = owner
                    owner = {
                        has_technology = "tech_transcendence_terraforming"
                    }
                }
                owner = {
                    add_resource = {
                        energy = 400
                        minerals = 600
                    }
                }
            }
        }
    }

    drop_weight = {
        weight = @blocker_medium_drop_weight
        modifier = {
            factor = 2
            is_planet_class = pc_desert
        }
        modifier = {
            factor = 0
            num_free_districts = {
                type = district_city
                value < 2
            }
        }
    }

    ai_weight = {
        weight = 1
    }

    prerequisites = {
        "tech_cleaning_molten_rock"
    }
}
