d_metacrystal_caves = {
    icon = d_metacrystal_caves

    is_for_colonizeable = yes
    category = deposit_cat_metamateria

    use_for_min_max_adjustments = yes
    use_weights_for_terraforming_swap_types = yes
    should_swap_deposit_on_terraforming = yes

    terraforming_swap_types = {
        d_submerged_ore_veins
        d_ore_rich_caverns
        d_mineral_fields
    }

    planet_modifier = {
        district_industrial_complex_max = 1
    }

    potential = {
        planet_size >= 10
        OR = {
            is_dry = yes
            is_planet_class = pc_nuked
        }
    }

    drop_weight = {
        weight = @blocker_ultralow_drop_weight
        modifier = {
            factor = 2
            is_planet_class = pc_desert
        }
        modifier = {
            factor = 3
            is_planet_class = pc_nuked
        }
    }
}
