resources_mithril_discovering_category = {
    desc = resources_mithril_discovering_category_desc
    picture = GFX_aoe_metallic_resources_discovering
    max_once_global = no
    max_once = no
    level = 3

    spawn_chance = {
        modifier = {
            add = 3
            has_mithril_resource = yes
        }
        modifier = {
            factor = 5
            from.owner = {
                NOT = { has_technology = "tech_mine_mithril" }
            }
        }
    }

    on_success = {
        1 = {
            ship_event = "aoe_ship_actions.10"
        }
    }
}
